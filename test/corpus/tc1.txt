{ transitive closure: powers R^1, R^2, R^3, R^4, ... }

TC1(R)
  DECL P,Q,S
  BEG
    P = R               { see RTC1; start with R instead of I }
    Q = O(R)
    S = P
    WHILE -eq(Q,S) DO
      P = P * R
      Q = S
      S = S | P
    OD
    RETURN S
  END.
