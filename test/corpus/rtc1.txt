{ Reflexive-Transitive Closure }
{ simple iteration computing the powers R^0, R^1, R^2, R^3, ... }

RTC1(R)
  DECL P,Q,S
  BEG
    P = I(R)            { P = current power of R }
    Q = O(R)            { Q = previous value of S }
    S = P               { S = sum of powers of R up to P }
    WHILE -eq(Q,S) DO
      P = P * R
      Q = S
      S = S | P
    OD
    RETURN S
  END.
