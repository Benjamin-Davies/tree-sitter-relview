RTC5
==================

{ Warshall's algorithm with heterogeneous vector }

RTC5(R)
  DECL P,Q,S
  BEG
    S = R
    Q = Ln1(R)                     { Q = nodes not yet considered }
    WHILE -empty(Q) DO
      P = point(Q)                 { select a node P }
      S = S | (S * P) * (P^ * S)   { use P as intermediate node }
      Q = Q & -P
    OD
    RETURN S | I(R)
  END.
