{ repeated squaring }

RTC2(R)
  DECL Q,S
  BEG
    Q = O(R)            { Q = previous value of S }
    S = I(R) | R        { S = sum of powers of R }
    WHILE -eq(Q,S) DO
      Q = S
      S = S * S
    OD
    RETURN S
  END.
