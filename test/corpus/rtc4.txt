{ Warshall's algorithm with partial identities }

RTC4(R)
  DECL P,Q,S
  BEG
    S = R
    Q = I(R)
    WHILE -empty(Q) DO
      P = atom(Q)
      S = S | S * P * S
      Q = Q & -P
    OD
    RETURN S | I(R)
  END.
