==================
Reachable1
==================

{ Nodes reachable in R from a set S of nodes }
{ S and result are vectors }

Reachable1(R,S)
  DECL Q
  BEG
    Q = S                          { Q = nodes reached so far }
    WHILE -empty(-Q & R^ * Q) DO   { any new nodes in next step? }
      Q = Q | R^ * Q               { add nodes reached in next step }
    OD
    RETURN Q
  END.
